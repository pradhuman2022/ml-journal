from sklearn.decomposition import PCA
from sklearn.preprocessing import RobustScaler

# 1. Separate features from the target species (using your df_features from earlier)
# Make sure to only include numeric columns for PCA
X = df_features.drop(columns=["Species"])
y = df_features["Species"]

# 2. PCA is highly sensitive to variances, so standardizing the data is required
X_scaled = RobustScaler().fit_transform(X)

# 3. Apply PCA to project down to 2 dimensions
pca = PCA(n_components=2)
X_pca = pca.fit_transform(X_scaled)

# 4. Create a new DataFrame with the principal components
df_pca = pd.DataFrame(data=X_pca, columns=["PC1", "PC2"])
df_pca["Species"] = y.values

# 5. Plot the 2D representation
plt.figure(figsize=(9, 6))
sns.scatterplot(
    data=df_pca,
    x="PC1",
    y="PC2",
    hue="Species",
    palette="dark",
    alpha=0.8,
    edgecolor="w",
    s=60,
)

# Calculate explained variance to show on the axes
var_exp = pca.explained_variance_ratio_ * 100
plt.xlabel(f"Principal Component 1 ({var_exp[0]:.1f}%)")
plt.ylabel(f"Principal Component 2 ({var_exp[1]:.1f}%)")
plt.title("Iris Dataset PCA - 2D Projection & Species Boundaries")
plt.grid(True, linestyle="--", alpha=0.5)
plt.show()

Learning Curve and Data Split Strategy¶

Data Split Strategy¶

A machine learning dataset is usually divided into multiple sets:

        Full Dataset
             |
-----------------------------
|            |              |

Train Validation Test

1. Training Set¶

  • Used to train the model.
  • Model learns patterns from this data.
  • Example: learn weights/parameters.

2. Validation Set¶

  • Used to compare models and tune parameters.
  • Helps choose:
    • model type
    • hyperparameters
    • PCA components
    • feature settings

3. Test Set¶

  • Used only at the end.
  • Measures final performance on unseen data.
  • Should not be used for model selection.

Cross Validation¶

Instead of keeping a separate validation set, we can use cross-validation on training data.

Example: 5-fold cross-validation

Dataset:

[A][B][C][D][E]

Round 1: Train: B,C,D,E Validate: A

Round 2: Train: A,C,D,E Validate: B

Round 3: Train: A,B,D,E Validate: C

Round 4: Train: A,B,C,E Validate: D

Round 5: Train: A,B,C,D Validate: E

Final validation score = average of all rounds.

Benefits:

  • More reliable performance estimate
  • Reduces dependency on one random split

3. Learning Curve¶

A learning curve shows how model performance changes as training data increases.

It trains the model on different dataset sizes: 10% data 20% data 30% data ... 100% data

For each size it measures:

  • Training score
  • Validation score

Example:

Data Size,Train Accuracy,Validation Accuracy 10%,99%,75% 50%,97%,88% 100%,96%,92%

4. Interpreting Learning Curve¶

Overfitting¶

Training Accuracy → High Validation Accuracy → Low

Meaning:

  • Model memorizes training data.
  • Does not generalize well.

Solutions:

  • More training data
  • Regularization
  • Simpler model

Underfitting¶

Training Accuracy → Low Validation Accuracy → Low

Meaning:

  • Model is too simple.

Solutions:

  • Better features
  • More complex model

Good Fit¶

Training Accuracy ≈ Validation Accuracy

Meaning:

  • Model learns patterns and generalizes well.

5. Correct ML Workflow¶

Raw Dataset | ↓ Split Data | ↓ Train / Validation / Test | ↓ Cross Validation + Learning Curve | ↓ Choose Model + Parameters | ↓ Train Final Model | ↓ Evaluate Once on Test Data

6. Important Rules¶

Never use test data for tuning¶

Test data should be the final unseen evaluation.

Fit preprocessing only on training data¶

Examples:

  • StandardScaler
  • PCA
  • Feature selection

Correct flow: Training Data | ↓ Fit Scaler | ↓ Fit PCA | ↓ Train Model

Validation/Test Data | ↓ Use same Scaler | ↓ Use same PCA | ↓ Prediction

7. Data Split Strategies¶

Random Split¶

Used when data points are independent.

Examples:

  • Classification
  • Regression
  • Tabular data

Example:

Train: 80% Test: 20%

Time-Based Split¶

Used when predicting future data.

Examples:

  • Stock prediction
  • Sales forecasting
  • Logs

Correct:

Train: Past data

Test: Future data

Do not randomly mix past and future.

Group-Based Split¶

Used when multiple rows belong to the same entity.

Examples:

  • Users
  • Patients
  • Devices

Avoid:

Train: User A record 1

Test: User A record 2

because it causes data leakage.

Correct:

Train: User A,B,C

Test: User D,E

Stratified Split¶

Used when classes are imbalanced.

Example:

Normal = 95% Fraud = 5%

Keep the same class ratio in train and test.

Final Summary¶

  • Training set → learns model parameters
  • Validation set → selects model and tuning
  • Test set → final evaluation only
  • Cross-validation → reliable validation inside training
  • Learning curve → checks effect of more data
  • PCA/Scaler should be fitted only on training data
  • Split strategy depends on the real-world problem

from sklearn.model_selection import learning_curve
from sklearn.linear_model import LogisticRegression

# 1. Define features and target using your PCA dataframe
X_pca_data = df_pca[["PC1", "PC2"]]
y_data = df_pca["Species"]

# 2. Set up the estimator
estimator = LogisticRegression()

# 3. Use the learning_curve function to automatically split, train, and score
train_sizes, train_scores, test_scores = learning_curve(
    estimator, 
    X_pca_data, 
    y_data, 
    cv=5,                             # 5-fold cross-validation splits
    scoring="accuracy", 
    train_sizes=np.linspace(0.1, 1.0, 10), # Test 10 different training set sizes
    shuffle=True,                     # <--- FIXES THE ONE-CLASS ERROR
    random_state=42,                  # <--- Keeps the random shuffle consistent
    n_jobs=-1,
)

# 4. Calculate mean and standard deviation for training and testing scores
train_mean = np.mean(train_scores, axis=1)
train_std = np.std(train_scores, axis=1)
test_mean = np.mean(test_scores, axis=1)
test_std = np.std(test_scores, axis=1)

# 5. Plot the learning curve
plt.figure(figsize=(10, 6))

# Plot training accuracy
plt.plot(train_sizes, train_mean, "o-", color="r", label="Training Score")
plt.fill_between(
    train_sizes, train_mean - train_std, train_mean + train_std, alpha=0.1, color="r"
)

# Plot cross-validation (testing) accuracy
plt.plot(
    train_sizes, test_mean, "o-", color="g", label="Cross-Validation Score"
)
plt.fill_between(
    train_sizes, test_mean - test_std, test_mean + test_std, alpha=0.1, color="g"
)

plt.title("Learning Curve (Logistic Regression on PCA Components)")
plt.xlabel("Training Examples Used")
plt.ylabel("Accuracy Score")
plt.legend(loc="best")
plt.grid(True, linestyle="--", alpha=0.5)
plt.show()

import matplotlib.pyplot as plt
import numpy as np
import pandas as pd
import seaborn as sns
from sklearn.preprocessing import LabelEncoder
from xgboost import XGBClassifier

# 1. Encode target string labels into integers (0, 1, 2)
le = LabelEncoder()
y_encoded = le.fit_transform(df_pca["Species"])
X_pca_data = df_pca[["PC1", "PC2"]].values

xgb_model = XGBClassifier(
    eval_metric="mlogloss",
    max_depth=3,          # Limit how deep each tree can grow (prevents micro-managing data)
    learning_rate=0.05,    # Slow down learning so it doesn't over-adjust to noise
    n_estimators=50,      # Reduce the total number of trees
    subsample=0.8,         # Train on a random 80% of rows per tree to add randomness
    colsample_bytree=0.8   # Train on a random 80% of features per tree
)
xgb_model.fit(X_pca_data, y_encoded)

# 3. Create a mesh grid to plot the decision boundary areas
x_min, x_max = X_pca_data[:, 0].min() - 1, X_pca_data[:, 0].max() + 1
y_min, y_max = X_pca_data[:, 1].min() - 1, X_pca_data[:, 1].max() + 1
xx, yy = np.meshgrid(np.arange(x_min, x_max, 0.02), np.arange(y_min, y_max, 0.02))

# 4. Predict the class for each point on the grid
grid_points = np.c_[xx.ravel(), yy.ravel()]
Z = xgb_model.predict(grid_points)
Z = Z.reshape(xx.shape)

# 5. Plot the background decision regions
plt.figure(figsize=(10, 7))
plt.contourf(xx, yy, Z, alpha=0.15, cmap=plt.cm.brg)

# Overlay the original scatter data points
sns.scatterplot(
    data=df_pca,
    x="PC1",
    y="PC2",
    hue="Species",
    palette="dark",
    alpha=0.9,
    edgecolor="w",
    s=70,
)

plt.xlabel("Principal Component 1")
plt.ylabel("Principal Component 2")
plt.title("XGBoost Decision Boundaries in PCA Space")
plt.grid(True, linestyle="--", alpha=0.3)
plt.show()

# 6. Print Model Accuracy
accuracy = xgb_model.score(X_pca_data, y_encoded) * 100
print(f"XGBoost Training Accuracy on PCA components: {accuracy:.2f}%")
